class Solution {
public:
    bool isValid(string s) {
        
        stack<char> sk;
        for(int i = 0;i < s.length();i++)
        {
            char ch = s[i];
            
            if(ch == '(' | ch == '{' | ch == '[')
            {
                sk.push(ch);
            }
            else
            {
                if(sk.empty())
                {
                    return false;
                }

                if(ch == ')')
                {
                    if(sk.top() != '(')
                    {
                
                        return false;
                    }
                    else
                    {
                        sk.pop();
                    }
                }

                if(ch == ']')
                {
                    if(sk.top() != '[')
                    {
                        return false;
                    }
                    else
                    {
                        sk.pop();
                    }
                }
                
                if(ch == '}')
                {
                    if(sk.top() != '{')
                    {
                        return false;
                    }
                    else
                    {
                        sk.pop();
                    }
                }		
            }
        }
	    return sk.empty();
    }
};
